```C
#include <stdio.h>
#include <string.h>
#pragma warning(disable:4996)

int delarr[9] = {};

//배열을 왼쪽으로 두칸 옮기는 함수
void delete_2()
{
	for (int i = 0; i < 7; i++)
	{
		delarr[i] = delarr[i + 2];
	}

}

int main()
{
	int arr[9] = {};
	int remainder = -100;	// 100 + 두수이기때문에, 나머지 두수의 합만 저장.
	int break_count = 0;

	for (int i = 0; i < 9; i++)
	{
		scanf("%d", &arr[i]);
		remainder += arr[i];
	}

	// printf("remainder : %d\n", remainder);

	for (int x = 0; x < 9; x++)
	{
		for (int y = x+1; y < 9; y++)
		{
			if (arr[x] + arr[y] == remainder)
			{	// 난쟁이가 아닌 두수에 대해 delarr의 제일 앞쪽 index(0, 1)에 저장
				delarr[0] = arr[x];	
				delarr[1] = arr[y];
				break_count++;
				break;
			}
		}

		if (break_count == 1)
			break;
	}

	// printf("%d %d\n", delarr[0], delarr[1]);
	
	int ctr = 2;

	for (int i = 0; i < 9; i++)
	{
		// 일곱 난쟁이 숫자들을 index 2번부터 저장
		if (arr[i] == delarr[0] || arr[i] == delarr[1])
		{
			continue;
		}

		delarr[ctr] = arr[i]; 
		ctr++;
	}

	delete_2(); // delarr의 0,1번 인덱스 제거
	
	//정렬
	for (int i = 0; i < 7; i++)
	{
		for (int j = i + 1; j < 7; j++)
		{
			if (delarr[j] < delarr[i])
			{
				int temp = delarr[i];
				delarr[i] = delarr[j];
				delarr[j] = temp;
			}
		}
	}

	for (int i = 0; i < 7; i++)
	{
		printf("%d\n", delarr[i]);
	}
}
```
